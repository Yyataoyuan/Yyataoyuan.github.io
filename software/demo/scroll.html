window.onload = function(){
function globalChange(){
    var index=0;
    var timer = setInterval(autoChange,3000);
    var pic_li = document.getElementById("banner_pic").getElementsByTagName("li");
    var button_li = document.getElementById("button").getElementsByTagName("li");

    for(var i=0; i<button_li.length; i++){
        button_li[i].onmouseover=function(){
            if(timer){clearInterval(timer);}
            for(var j=0; j<button_li.length;j++){
                if(button_li[j]==this){
                    index=j;
                    pic_li[j].className="show";
                    button_li[j].className="fore";
                }
                else{
                    pic_li[j].className="hidden";
                    button_li[j].className="back";
                }
            }
        }

        button_li[i].onmouseout=function(){
            
        }
    }

    function autoChange(){
        ++index;
        if(index==pic_li.length){
            index=0;
        }
        for(var i=0; i<pic_li.length;i++){
            if(index==i){
                pic_li[i].className="show";
                button_li[i].className="fore";
            }
            else{
                pic_li[i].className="hidden";
                button_li[i].className="back";
            }
        }
            }
        }
        globalChange();
    }
