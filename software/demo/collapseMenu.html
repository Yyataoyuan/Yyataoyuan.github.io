window.onload=function(){
    var arr_p = document.getElementsByTagName("p");
    var arr_ul = document.getElementsByTagName("ul");
    
    for(var i=0; i<arr_p.length; i++){
        arr_p[i].onclick = function(){
            for(var j=0; j<arr_ul.length; j++){
                // 如果数组arr_p中的元素被点击
                // 且数组arr_ul的display属性为block时把arr_ul的display设置为display
                if(arr_p[j]==this&&arr_ul[j].style.display != "block"){
                    arr_ul[j].style.display = "block";
                }
                else{
                    arr_ul[j].style.display = "none";
                }
            }
        }
    }
